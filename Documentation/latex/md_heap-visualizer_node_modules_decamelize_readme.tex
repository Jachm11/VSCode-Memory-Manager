\begin{quote}
Convert a camelized string into a lowercased one with a custom separator~\newline
 Example\+: {\ttfamily unicorn\+Rainbow} → {\ttfamily unicorn\+\_\+rainbow} \end{quote}


\doxysection*{Install}


\begin{DoxyCode}{0}
\DoxyCodeLine{\$ npm install -\/-\/save decamelize}
\end{DoxyCode}


\doxysection*{Usage}


\begin{DoxyCode}{0}
\DoxyCodeLine{const decamelize = require('decamelize');}
\DoxyCodeLine{}
\DoxyCodeLine{decamelize('unicornRainbow');}
\DoxyCodeLine{//=> 'unicorn\_rainbow'}
\DoxyCodeLine{}
\DoxyCodeLine{decamelize('unicornRainbow', '-\/');}
\DoxyCodeLine{//=> 'unicorn-\/rainbow'}
\end{DoxyCode}
