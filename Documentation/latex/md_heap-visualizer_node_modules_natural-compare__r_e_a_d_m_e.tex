\doxysection*{Natural Compare -- \href{https://travis-ci.org/litejs/natural-compare-lite}{\texttt{ }} \href{https://coveralls.io/r/litejs/natural-compare-lite}{\texttt{ }} }

Compare strings containing a mix of letters and numbers in the way a human being would in sort order. This is described as a \char`\"{}natural ordering\char`\"{}.


\begin{DoxyCode}{0}
\DoxyCodeLine{Standard sorting:   Natural order sorting:}
\DoxyCodeLine{    img1.png            img1.png}
\DoxyCodeLine{    img10.png           img2.png}
\DoxyCodeLine{    img12.png           img10.png}
\DoxyCodeLine{    img2.png            img12.png}
\end{DoxyCode}


String.\+natural\+Compare returns a number indicating whether a reference string comes before or after or is the same as the given string in sort order. Use it with builtin sort() function.

\doxysubsubsection*{Usage}


\begin{DoxyCode}{0}
\DoxyCodeLine{// Simple case sensitive example}
\DoxyCodeLine{var a = ["z1.doc", "z10.doc", "z17.doc", "z2.doc", "z23.doc", "z3.doc"];}
\DoxyCodeLine{a.sort(String.naturalCompare);}
\DoxyCodeLine{// ["z1.doc", "z2.doc", "z3.doc", "z10.doc", "z17.doc", "z23.doc"]}
\DoxyCodeLine{}
\DoxyCodeLine{// Use wrapper function for case insensitivity}
\DoxyCodeLine{a.sort(function(a, b)\{}
\DoxyCodeLine{  return String.naturalCompare(a.toLowerCase(), b.toLowerCase());}
\DoxyCodeLine{\})}
\DoxyCodeLine{}
\DoxyCodeLine{// In most cases we want to sort an array of objects}
\DoxyCodeLine{var a = [ \{"street":"350 5th Ave", "room":"A-\/1021"\}}
\DoxyCodeLine{        , \{"street":"350 5th Ave", "room":"A-\/21046-\/b"\} ];}
\DoxyCodeLine{}
\DoxyCodeLine{// sort by street, then by room}
\DoxyCodeLine{a.sort(function(a, b)\{}
\DoxyCodeLine{  return String.naturalCompare(a.street, b.street) || String.naturalCompare(a.room, b.room);}
\DoxyCodeLine{\})}
\DoxyCodeLine{}
\DoxyCodeLine{// When text transformation is needed (eg toLowerCase()),}
\DoxyCodeLine{// it is best for performance to keep}
\DoxyCodeLine{// transformed key in that object.}
\DoxyCodeLine{// There are no need to do text transformation}
\DoxyCodeLine{// on each comparision when sorting.}
\DoxyCodeLine{var a = [ \{"make":"Audi", "model":"A6"\}}
\DoxyCodeLine{        , \{"make":"Kia",  "model":"Rio"\} ];}
\DoxyCodeLine{}
\DoxyCodeLine{// sort by make, then by model}
\DoxyCodeLine{a.map(function(car)\{}
\DoxyCodeLine{  car.sort\_key = (car.make + " " + car.model).toLowerCase();}
\DoxyCodeLine{\})}
\DoxyCodeLine{a.sort(function(a, b)\{}
\DoxyCodeLine{  return String.naturalCompare(a.sort\_key, b.sort\_key);}
\DoxyCodeLine{\})}
\end{DoxyCode}

\doxysubsubsection*{Custom alphabet}

It is possible to configure a custom alphabet to achieve a desired order.


\begin{DoxyCode}{0}
\DoxyCodeLine{// Estonian alphabet}
\DoxyCodeLine{String.alphabet = "ABDEFGHIJKLMNOPRSŠZŽTUVÕÄÖÜXYabdefghijklmnoprsšzžtuvõäöüxy"}
\DoxyCodeLine{["t", "z", "x", "õ"].sort(String.naturalCompare)}
\DoxyCodeLine{// ["z", "t", "õ", "x"]}
\DoxyCodeLine{}
\DoxyCodeLine{// Russian alphabet}
\DoxyCodeLine{String.alphabet = "АБВГДЕЁЖЗИЙКЛМНОПРСТУФХЦЧШЩЪЫЬЭЮЯабвгдеёжзийклмнопрстуфхцчшщъыьэюя"}
\DoxyCodeLine{["Ё", "А", "Б"].sort(String.naturalCompare)}
\DoxyCodeLine{// ["А", "Б", "Ё"]}
\end{DoxyCode}
