\href{https://npmjs.org/package/has-symbols}{\texttt{ }}

Determine if the JS environment has Symbol support. Supports spec, or shams.

\doxysection*{Example}


\begin{DoxyCode}{0}
\DoxyCodeLine{var hasSymbols = require('has-\/symbols');}
\DoxyCodeLine{}
\DoxyCodeLine{hasSymbols() === true; // if the environment has native Symbol support. Not polyfillable, not forgeable.}
\DoxyCodeLine{}
\DoxyCodeLine{var hasSymbolsKinda = require('has-\/symbols/shams');}
\DoxyCodeLine{hasSymbolsKinda() === true; // if the environment has a Symbol sham that mostly follows the spec.}
\end{DoxyCode}
