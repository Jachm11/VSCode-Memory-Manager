Is this JS value callable? Works with Functions and Generator\+Functions, despite E\+S6 @to\+String\+Tag.

\doxysection*{Example}


\begin{DoxyCode}{0}
\DoxyCodeLine{var isCallable = require('is-\/callable');}
\DoxyCodeLine{var assert = require('assert');}
\DoxyCodeLine{}
\DoxyCodeLine{assert.notOk(isCallable(undefined));}
\DoxyCodeLine{assert.notOk(isCallable(null));}
\DoxyCodeLine{assert.notOk(isCallable(false));}
\DoxyCodeLine{assert.notOk(isCallable(true));}
\DoxyCodeLine{assert.notOk(isCallable([]));}
\DoxyCodeLine{assert.notOk(isCallable(\{\}));}
\DoxyCodeLine{assert.notOk(isCallable(/a/g));}
\DoxyCodeLine{assert.notOk(isCallable(new RegExp('a', 'g')));}
\DoxyCodeLine{assert.notOk(isCallable(new Date()));}
\DoxyCodeLine{assert.notOk(isCallable(42));}
\DoxyCodeLine{assert.notOk(isCallable(NaN));}
\DoxyCodeLine{assert.notOk(isCallable(Infinity));}
\DoxyCodeLine{assert.notOk(isCallable(new Number(42)));}
\DoxyCodeLine{assert.notOk(isCallable('foo'));}
\DoxyCodeLine{assert.notOk(isCallable(Object('foo')));}
\DoxyCodeLine{}
\DoxyCodeLine{assert.ok(isCallable(function () \{\}));}
\DoxyCodeLine{assert.ok(isCallable(function* () \{\}));}
\DoxyCodeLine{assert.ok(isCallable(x => x * x));}
\end{DoxyCode}
