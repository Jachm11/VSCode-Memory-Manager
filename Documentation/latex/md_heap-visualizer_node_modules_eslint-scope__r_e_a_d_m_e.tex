E\+S\+Lint Scope is the \href{http://www.ecma-international.org/publications/standards/Ecma-262.htm}{\texttt{ E\+C\+M\+A\+Script}} scope analyzer used in E\+S\+Lint. It is a fork of \href{http://github.com/estools/escope}{\texttt{ escope}}.

\doxysection*{Usage}

Install\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{npm i eslint-\/scope -\/-\/save}
\end{DoxyCode}


Example\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{var eslintScope = require('eslint-\/scope');}
\DoxyCodeLine{var espree = require('espree');}
\DoxyCodeLine{var estraverse = require('estraverse');}
\DoxyCodeLine{}
\DoxyCodeLine{var ast = espree.parse(code);}
\DoxyCodeLine{var scopeManager = eslintScope.analyze(ast);}
\DoxyCodeLine{}
\DoxyCodeLine{var currentScope = scopeManager.acquire(ast);   // global scope}
\DoxyCodeLine{}
\DoxyCodeLine{estraverse.traverse(ast, \{}
\DoxyCodeLine{    enter: function(node, parent) \{}
\DoxyCodeLine{        // do stuff}
\DoxyCodeLine{}
\DoxyCodeLine{        if (/Function/.test(node.type)) \{}
\DoxyCodeLine{            currentScope = scopeManager.acquire(node);  // get current function scope}
\DoxyCodeLine{        \}}
\DoxyCodeLine{    \},}
\DoxyCodeLine{    leave: function(node, parent) \{}
\DoxyCodeLine{        if (/Function/.test(node.type)) \{}
\DoxyCodeLine{            currentScope = currentScope.upper;  // set to parent scope}
\DoxyCodeLine{        \}}
\DoxyCodeLine{}
\DoxyCodeLine{        // do stuff}
\DoxyCodeLine{    \}}
\DoxyCodeLine{\});}
\end{DoxyCode}
