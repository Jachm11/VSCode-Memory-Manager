Deterministic {\ttfamily J\+S\+O\+N.\+stringify()} -\/ a faster version of \href{https://github.com/substack}{\texttt{ @substack}}\textquotesingle{}s json-\/stable-\/strigify without \href{https://github.com/substack/jsonify}{\texttt{ jsonify}}.

You can also pass in a custom comparison function.

\href{https://travis-ci.org/epoberezkin/fast-json-stable-stringify}{\texttt{ }} \href{https://coveralls.io/github/epoberezkin/fast-json-stable-stringify?branch=master}{\texttt{ }}

\doxysection*{example}


\begin{DoxyCode}{0}
\DoxyCodeLine{var stringify = require('fast-\/json-\/stable-\/stringify');}
\DoxyCodeLine{var obj = \{ c: 8, b: [\{z:6,y:5,x:4\},7], a: 3 \};}
\DoxyCodeLine{console.log(stringify(obj));}
\end{DoxyCode}


output\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{\{"a":3,"b":[\{"x":4,"y":5,"z":6\},7],"c":8\}}
\end{DoxyCode}

\doxysubsection*{var str = stringify(obj, opts)}

Return a deterministic stringified string {\ttfamily str} from the object {\ttfamily obj}.

\doxysubsection*{options}

\doxysubsubsection*{cmp}

If {\ttfamily opts} is given, you can supply an {\ttfamily opts.\+cmp} to have a custom comparison function for object keys. Your function {\ttfamily opts.\+cmp} is called with these parameters\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{opts.cmp(\{ key: akey, value: avalue \}, \{ key: bkey, value: bvalue \})}
\end{DoxyCode}


For example, to sort on the object key names in reverse order you could write\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{var stringify = require('fast-\/json-\/stable-\/stringify');}
\DoxyCodeLine{}
\DoxyCodeLine{var obj = \{ c: 8, b: [\{z:6,y:5,x:4\},7], a: 3 \};}
\DoxyCodeLine{var s = stringify(obj, function (a, b) \{}
\DoxyCodeLine{    return a.key < b.key ? 1 : -\/1;}
\DoxyCodeLine{\});}
\DoxyCodeLine{console.log(s);}
\end{DoxyCode}


which results in the output string\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{\{"c":8,"b":[\{"z":6,"y":5,"x":4\},7],"a":3\}}
\end{DoxyCode}


Or if you wanted to sort on the object values in reverse order, you could write\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{var stringify = require('fast-\/json-\/stable-\/stringify');}
\DoxyCodeLine{}
\DoxyCodeLine{var obj = \{ d: 6, c: 5, b: [\{z:3,y:2,x:1\},9], a: 10 \};}
\DoxyCodeLine{var s = stringify(obj, function (a, b) \{}
\DoxyCodeLine{    return a.value < b.value ? 1 : -\/1;}
\DoxyCodeLine{\});}
\DoxyCodeLine{console.log(s);}
\end{DoxyCode}


which outputs\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{\{"d":6,"c":5,"b":[\{"z":3,"y":2,"x":1\},9],"a":10\}}
\end{DoxyCode}


\doxysubsubsection*{cycles}

Pass {\ttfamily true} in {\ttfamily opts.\+cycles} to stringify circular property as {\ttfamily \+\_\+\+\_\+cycle\+\_\+\+\_\+} -\/ the result will not be a valid J\+S\+ON string in this case.

Type\+Error will be thrown in case of circular object without this option.
