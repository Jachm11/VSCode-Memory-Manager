\begin{quote}
Get the path of the parent module \end{quote}


Node.\+js exposes {\ttfamily module.\+parent}, but it only gives you the first cached parent, which is not necessarily the actual parent.

\doxysection*{Install}


\begin{DoxyCode}{0}
\DoxyCodeLine{\$ npm install parent-\/module}
\end{DoxyCode}


\doxysection*{Usage}


\begin{DoxyCode}{0}
\DoxyCodeLine{// bar.js}
\DoxyCodeLine{const parentModule = require('parent-\/module');}
\DoxyCodeLine{}
\DoxyCodeLine{module.exports = () => \{}
\DoxyCodeLine{    console.log(parentModule());}
\DoxyCodeLine{    //=> '/Users/sindresorhus/dev/unicorn/foo.js'}
\DoxyCodeLine{\};}
\end{DoxyCode}



\begin{DoxyCode}{0}

\DoxyCodeLine{const bar = require('./bar');}

\doxysection*{A\+PI}

\doxysubsection*{parent\+Module(\mbox{[}filepath\mbox{]})}

By default, it will return the path of the immediate parent.

\doxysubsubsection*{filepath}

Type\+: {\ttfamily string}~\newline
 Default\+: \href{https://nodejs.org/api/globals.html\#globals_filename}{\texttt{ {\ttfamily \+\_\+\+\_\+filename}}}

Filepath of the module of which to get the parent path.

Useful if you want it to work \href{https://github.com/sindresorhus/parent-module/tree/master/fixtures/filepath}{\texttt{ multiple module levels down}}.

\doxysection*{Tip}

Combine it with \href{https://github.com/sindresorhus/read-pkg-up}{\texttt{ {\ttfamily read-\/pkg-\/up}}} to read the package.\+json of the parent module.


\begin{DoxyCode}{0}
\DoxyCodeLine{const path = require('path');}
\DoxyCodeLine{const readPkgUp = require('read-\/pkg-\/up');}
\DoxyCodeLine{const parentModule = require('parent-\/module');}
\DoxyCodeLine{}
\DoxyCodeLine{console.log(readPkgUp.sync(\{cwd: path.dirname(parentModule())\}).pkg);}
\DoxyCodeLine{//=> \{name: 'chalk', version: '1.0.0', …\}}
\end{DoxyCode}
