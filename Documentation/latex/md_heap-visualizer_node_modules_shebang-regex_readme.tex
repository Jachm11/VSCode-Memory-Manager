\begin{quote}
Regular expression for matching a \href{https://en.wikipedia.org/wiki/Shebang_(Unix)}{\texttt{ shebang}} \end{quote}


\doxysection*{Install}


\begin{DoxyCode}{0}
\DoxyCodeLine{\$ npm install -\/-\/save shebang-\/regex}
\end{DoxyCode}


\doxysection*{Usage}


\begin{DoxyCode}{0}
\DoxyCodeLine{var shebangRegex = require('shebang-\/regex');}
\DoxyCodeLine{var str = '\#!/usr/bin/env node\(\backslash\)nconsole.log("unicorns");';}
\DoxyCodeLine{}
\DoxyCodeLine{shebangRegex.test(str);}
\DoxyCodeLine{//=> true}
\DoxyCodeLine{}
\DoxyCodeLine{shebangRegex.exec(str)[0];}
\DoxyCodeLine{//=> '\#!/usr/bin/env node'}
\end{DoxyCode}
