\begin{quote}
Get the visual width of a string -\/ the number of columns required to display it \end{quote}


Some Unicode characters are \href{https://en.wikipedia.org/wiki/Halfwidth_and_fullwidth_forms}{\texttt{ fullwidth}} and use double the normal width. \href{https://en.wikipedia.org/wiki/ANSI_escape_code}{\texttt{ A\+N\+SI escape codes}} are stripped and doesn\textquotesingle{}t affect the width.

Useful to be able to measure the actual width of command-\/line output.

\doxysection*{Install}


\begin{DoxyCode}{0}
\DoxyCodeLine{\$ npm install string-\/width}
\end{DoxyCode}


\doxysection*{Usage}


\begin{DoxyCode}{0}
\DoxyCodeLine{const stringWidth = require('string-\/width');}
\DoxyCodeLine{}
\DoxyCodeLine{stringWidth('古');}
\DoxyCodeLine{//=> 2}
\DoxyCodeLine{}
\DoxyCodeLine{stringWidth('\(\backslash\)u001b[1m古\(\backslash\)u001b[22m');}
\DoxyCodeLine{//=> 2}
\DoxyCodeLine{}
\DoxyCodeLine{stringWidth('a');}
\DoxyCodeLine{//=> 1}
\DoxyCodeLine{}
\DoxyCodeLine{stringWidth('\(\backslash\)u001B]8;;https://github.com\(\backslash\)u0007Click\(\backslash\)u001B]8;;\(\backslash\)u0007');}
\DoxyCodeLine{// => 5}
\end{DoxyCode}
