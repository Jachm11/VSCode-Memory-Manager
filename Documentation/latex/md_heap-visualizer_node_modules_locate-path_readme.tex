\begin{quote}
Get the first path that exists on disk of multiple paths \end{quote}


\doxysection*{Install}


\begin{DoxyCode}{0}
\DoxyCodeLine{\$ npm install locate-\/path}
\end{DoxyCode}


\doxysection*{Usage}

Here we find the first file that exists on disk, in array order.


\begin{DoxyCode}{0}
\DoxyCodeLine{const locatePath = require('locate-\/path');}
\DoxyCodeLine{}
\DoxyCodeLine{const files = [}
\DoxyCodeLine{    'unicorn.png',}
\DoxyCodeLine{    'rainbow.png', // Only this one actually exists on disk}
\DoxyCodeLine{    'pony.png'}
\DoxyCodeLine{];}
\DoxyCodeLine{}
\DoxyCodeLine{(async () => \{}
\DoxyCodeLine{    console(await locatePath(files));}
\DoxyCodeLine{    //=> 'rainbow'}
\DoxyCodeLine{\})();}
\end{DoxyCode}

\doxysubsection*{locate\+Path(input, \mbox{[}options\mbox{]})}

Returns a {\ttfamily Promise} for the first path that exists or {\ttfamily undefined} if none exists.

\doxysubsubsection*{input}

Type\+: {\ttfamily Iterable$<$string$>$}
