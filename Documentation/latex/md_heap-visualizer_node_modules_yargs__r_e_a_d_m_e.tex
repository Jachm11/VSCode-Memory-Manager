\doxysubsection*{Description \+:}

Yargs helps you build interactive command line tools, by parsing arguments and generating an elegant user interface.

It gives you\+:


\begin{DoxyItemize}
\item commands and (grouped) options ({\ttfamily my-\/program.\+js serve -\/-\/port=5000}).
\item a dynamically generated help menu based on your arguments.
\end{DoxyItemize}

\begin{quote}
 \end{quote}



\begin{DoxyItemize}
\item bash-\/completion shortcuts for commands and options.
\item and tons more.
\end{DoxyItemize}

\doxysubsection*{Usage \+:}

\doxysubsubsection*{Simple Example}


\begin{DoxyCode}{0}
\DoxyCodeLine{\#!/usr/bin/env node}
\DoxyCodeLine{const argv = require('yargs').argv}
\DoxyCodeLine{}
\DoxyCodeLine{if (argv.ships > 3 \&\& argv.distance < 53.5) \{}
\DoxyCodeLine{  console.log('Plunder more riffiwobbles!')}
\DoxyCodeLine{\} else \{}
\DoxyCodeLine{  console.log('Retreat from the xupptumblers!')}
\DoxyCodeLine{\}}
\end{DoxyCode}



\begin{DoxyCode}{0}
\DoxyCodeLine{\$ ./plunder.js -\/-\/ships=4 -\/-\/distance=22}
\DoxyCodeLine{Plunder more riffiwobbles!}
\DoxyCodeLine{}
\DoxyCodeLine{\$ ./plunder.js -\/-\/ships 12 -\/-\/distance 98.7}
\DoxyCodeLine{Retreat from the xupptumblers!}
\end{DoxyCode}


\doxysubsubsection*{Complex Example}


\begin{DoxyCode}{0}
\DoxyCodeLine{\#!/usr/bin/env node}
\DoxyCodeLine{require('yargs') // eslint-\/disable-\/line}
\DoxyCodeLine{  .command('serve [port]', 'start the server', (yargs) => \{}
\DoxyCodeLine{    yargs}
\DoxyCodeLine{      .positional('port', \{}
\DoxyCodeLine{        describe: 'port to bind on',}
\DoxyCodeLine{        default: 5000}
\DoxyCodeLine{      \})}
\DoxyCodeLine{  \}, (argv) => \{}
\DoxyCodeLine{    if (argv.verbose) console.info(`start server on :\$\{argv.port\}`)}
\DoxyCodeLine{    serve(argv.port)}
\DoxyCodeLine{  \})}
\DoxyCodeLine{  .option('verbose', \{}
\DoxyCodeLine{    alias: 'v',}
\DoxyCodeLine{    default: false}
\DoxyCodeLine{  \})}
\DoxyCodeLine{  .argv}
\end{DoxyCode}


Run the example above with {\ttfamily -\/-\/help} to see the help for the application.
