\begin{quote}
Check if \href{https://nodejs.org/docs/latest/api/process.html\#process_process_argv}{\texttt{ {\ttfamily argv}}} has a specific flag \end{quote}


Correctly stops looking after an {\ttfamily -\/-\/} argument terminator.

\doxysection*{Install}


\begin{DoxyCode}{0}
\DoxyCodeLine{\$ npm install has-\/flag}
\end{DoxyCode}


\doxysection*{Usage}


\begin{DoxyCode}{0}
\DoxyCodeLine{// foo.js}
\DoxyCodeLine{const hasFlag = require('has-\/flag');}
\DoxyCodeLine{}
\DoxyCodeLine{hasFlag('unicorn');}
\DoxyCodeLine{//=> true}
\DoxyCodeLine{}
\DoxyCodeLine{hasFlag('-\/-\/unicorn');}
\DoxyCodeLine{//=> true}
\DoxyCodeLine{}
\DoxyCodeLine{hasFlag('f');}
\DoxyCodeLine{//=> true}
\DoxyCodeLine{}
\DoxyCodeLine{hasFlag('-\/f');}
\DoxyCodeLine{//=> true}
\DoxyCodeLine{}
\DoxyCodeLine{hasFlag('foo=bar');}
\DoxyCodeLine{//=> true}
\DoxyCodeLine{}

\DoxyCodeLine{}
\DoxyCodeLine{hasFlag('rainbow');}
\DoxyCodeLine{//=> false}
\end{DoxyCode}

\doxysection*{A\+PI}

\doxysubsection*{has\+Flag(flag, \mbox{[}argv\mbox{]})}

Returns a boolean for whether the flag exists.

\doxysubsubsection*{flag}

Type\+: {\ttfamily string}

C\+LI flag to look for. The {\ttfamily -\/-\/} prefix is optional.

\doxysubsubsection*{argv}

Type\+: {\ttfamily string\mbox{[}\mbox{]}}~\newline
 Default\+: {\ttfamily process.\+argv}

C\+LI arguments.
